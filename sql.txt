create table hop.beer_brand_tally_sent(
    beer_id  int identity PRIMARY KEY, 
    brand varchar(30) not null, 
    volume int not null,
    neg FLOAT not null,
    neu FLOAT not null,
    pos FLOAT not null,
    compound FLOAT not null, 
    created_datetime DATETIME DEFAULT(GETDATE()),
    );

    create table hop.beer_style_tally_sent(
    beer_id  int identity PRIMARY KEY, 
    style varchar(30) not null, 
    volume int not null,
    neg FLOAT not null,
    neu FLOAT not null,
    pos FLOAT not null,
    compound FLOAT not null, 
    created_datetime DATETIME DEFAULT(GETDATE()),
    );
